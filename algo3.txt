PROCEDURE dot_Product(T1,T2: ARRAY_OF FLOAT[3], VAR ps: FLOAT )
VAR
    i,j : INTEGER:=0,
BEGIN
    FOR i FROM 0 TO n-1 STEP  DO
        FOR j FROM 0 TO n-1 STEP step  DO
            ps := ps + T1[i]*T2[j]
        END_FOR
    END_FOR
END
